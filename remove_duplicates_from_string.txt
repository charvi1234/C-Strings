#include<string.h>
void remDuplicates(char *str)
{
int arr[100];
  char str2[1000];
  int i,k=0;
  for(i=0;i<26;i++)
    arr[i]=0;
  int len=strlen(str);
  for(i=0;i<len;i++)
  {
    if(str[i]>='a'&&str[i]<='z')
      arr[str[i]-97]++;
  }
  for(i=0;i<26;i++)
  {
    if(arr[i]>=1)
    {
      str2[k]=i+97;
      k++;
    }
  }
  str2[k]='\0';
  strcpy(str,str2);
}
